=title Table Record Design A (tableA)

The tableA record stores a multi-column table. Column metadata (names and
types) is set at initialization; data is written per-column via COL00..COL07
pseudo-fields. qsrv publishes this record as an NTTable.

=recordtype tableA

=cut

recordtype(tableA) {
    include "dbCommon.dbd"

    %
    %/* Device Support Entry Table */
    %struct tableARecord;
    %typedef struct {
    %    dset common;
    %    long (*read_table)(struct tableARecord *prec);
    %} tableAdset;
    %#define HAS_tableAdset
    %

    field(VAL, DBF_NOACCESS) {
        prompt("Value (alias for COL00)")
        asl(ASL0)
        special(SPC_DBADDR)
        pp(TRUE)
        extra("void *val")
    }
    field(INP, DBF_INLINK) {
        prompt("Input Specification")
        promptgroup("40 - Input")
        interest(1)
    }
    field(MAXCOLS, DBF_ULONG) {
        prompt("Max number of columns")
        special(SPC_NOMOD)
        interest(1)
        initial("8")
    }
    field(MAXROWS, DBF_ULONG) {
        prompt("Max number of rows")
        special(SPC_NOMOD)
        interest(1)
        initial("16")
    }
    field(NCOL, DBF_ULONG) {
        prompt("Number of active columns")
        interest(1)
    }
    field(NROW, DBF_ULONG) {
        prompt("Number of active rows")
        interest(1)
    }
    field(COLNAMES, DBF_NOACCESS) {
        prompt("Column names (STRING[MAXCOLS])")
        special(SPC_DBADDR)
        extra("char *colnames")
        interest(1)
    }
    field(COLTYPES, DBF_NOACCESS) {
        prompt("Column types — menuFtype values (USHORT[MAXCOLS])")
        special(SPC_DBADDR)
        extra("epicsUInt16 *coltypes")
        interest(1)
    }
    field(CHGD, DBF_NOACCESS) {
        prompt("Column changed flags (UCHAR[MAXCOLS])")
        special(SPC_DBADDR)
        extra("epicsUInt8 *chgd")
        interest(1)
    }
    field(COL00, DBF_NOACCESS) {
        prompt("Column 0 data")
        special(SPC_DBADDR)
        extra("void *col00")
        interest(2)
    }
    field(COL01, DBF_NOACCESS) {
        prompt("Column 1 data")
        special(SPC_DBADDR)
        extra("void *col01")
        interest(2)
    }
    field(COL02, DBF_NOACCESS) {
        prompt("Column 2 data")
        special(SPC_DBADDR)
        extra("void *col02")
        interest(2)
    }
    field(COL03, DBF_NOACCESS) {
        prompt("Column 3 data")
        special(SPC_DBADDR)
        extra("void *col03")
        interest(2)
    }
    field(COL04, DBF_NOACCESS) {
        prompt("Column 4 data")
        special(SPC_DBADDR)
        extra("void *col04")
        interest(2)
    }
    field(COL05, DBF_NOACCESS) {
        prompt("Column 5 data")
        special(SPC_DBADDR)
        extra("void *col05")
        interest(2)
    }
    field(COL06, DBF_NOACCESS) {
        prompt("Column 6 data")
        special(SPC_DBADDR)
        extra("void *col06")
        interest(2)
    }
    field(COL07, DBF_NOACCESS) {
        prompt("Column 7 data")
        special(SPC_DBADDR)
        extra("void *col07")
        interest(2)
    }
}
